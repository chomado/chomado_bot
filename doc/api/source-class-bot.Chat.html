 31     /**
 32      * docomoの対話APIを叩いてレスポンスを貰ってくる
 33      *
 34      * @param string $apikey    docomoAPIキー
 35      * @param string $context   会話のコンテキストID(API仕様参照)
 36      * @param string $mode      会話のモード(API仕様参照
 37      * @param string $nickname  会話している人間側の名前
 38      * @param string $text      人間側の入力テキスト
 39      * @return object レスポンスのJSONをデコードしたオブジェクト
 40      */
 41     private function GetData($apikey, $context, $mode, $nickname, $text)
 42     {
 43         $user_data = [
 44             'utt'       => self::sjisSafe((string)$text),
 45             'context'   => (string)$context,
 46             'nickname'  => self::sjisSafe((string)$nickname),
 47             'mode'      => (string)$mode,
 48         ];
 49         $url = sprintf(
 50             'https://api.apigw.smt.docomo.ne.jp/dialogue/v1/dialogue?APIKEY=%s',
 51             rawurlencode($apikey)
 52         );
 53         $options = [
 54             'http' => [
 55                 'method'    => 'POST',
 56                 'header'    => "Content-type: application/json",
 57                 'content'   => json_encode($user_data),
 58             ],
 59         ];
 60 
 61         Log::info("docomo対話APIを呼び出します");
 62         Log::info("URL: " . $url);
 63         Log::info("パラメータ:");
 64         Log::info($user_data);
 65         
 66         $response = file_get_contents($url, false, stream_context_create($options));
 67         $ret = @json_decode($response);
 68         if($response === false) {
 69             Log::error("docomo対話APIの呼び出しに失敗しました");
 70         } else {
 71             Log::info("docomoからのデータ:");
 72             Log::info($response);
 73             if(is_object($ret) && isset($ret->utt) && $ret->utt != '') {
 74                 Log::success("  docomo 指示文章: " . $ret->utt);
 75             }
 76         }
 77         return $ret;
 78     }

109     /**
110      * 与えられた文字列を Shift-JIS で収まる範囲に変換する
111      *
112      * docomo APIがどうも Shift-JIS で収まらないものを与えると Bad Request で死ぬようなので
113      * とりあえず SJIS で収まるもののみを送るようにしてみる
114      *
115      * @param string $text 変換対象文字列
116      * @return string
117      */
118     private static function sjisSafe($text) {
119         mb_substitute_character(0x3013); // 変換できない文字をゲタにする
120         return mb_convert_encoding(
121             mb_convert_encoding($text, 'CP932', 'UTF-8'),
122             'UTF-8',
123             'CP932'
124         );
125     }
