 58     /**
 59      * {@inheritdoc}
 60      */
 61     public function write($data, $level) {
 62         $now = time();
 63         $int_level = $this->convertStringErrorLevelToInt($level);
 64         $min_level = $this->getMinLogLevel();
 65         $max_level = $this->getMaxLogLevel();
 66         if($min_level <= $int_level && $int_level <= $max_level) {
 67             if($data === null) { // is_scalar は NULL をスカラ扱いしない
 68                 $text = '(NULL)';
 69             } elseif($data === true) { // boolean をそのまま stringify すると "0"/"1" で誰も幸せにならない
 70                 $text = '(TRUE)';
 71             } elseif($data === false) {
 72                 $text = '(FALSE)';
 73             } elseif(is_resource($data)) { // is_scalarはリソース型にfalseを返すがこの挙動に依存するなと書いてあるので先に判定
 74                 $text = '(RESOURCE)';
 75             } elseif(is_scalar($data)) { // integer, float, string
 76                 $text = (string)$data;
 77             } elseif(is_object($data) && is_callable([$data, '__toString'])) {
 78                 $text = $data->__toString();
 79             } else {
 80                 $text = '';
 81                 if(is_object($data)) {
 82                     $text = '<class:' . get_class($data) . '> ';
 83                 }
 84                 $json = @json_encode($data, JSON_PRETTY_PRINT|JSON_UNESCAPED_UNICODE|JSON_UNESCAPED_SLASHES);
 85                 $text .= ($json === false) ? '(ERROR)' : $json;
 86             }
 87 
 88             // 実装クラスが楽になるように複数行にまたがるものはあらかじめ各行に分割してから送る
 89             foreach(preg_split('/\x0d\x0a|\x0d|\x0a/', $text) as $line) {
 90                 $this->writeImpl($now, $line, $level, $int_level);
 91             }
 92         }
 93     }
