 92     /**

 99     private function decorate($text, $int_level) {
100         $colors = isset(self::$error_level_color_map[$int_level]) ? self::$error_level_color_map[$int_level] : null;
101         if(!$colors) {
102             return $text;
103         }
104 
105         $ret = '';
106         // 文字色
107         if(isset(self::$foreground_colors[$colors[0]])) {
108             $ret .= "\033[" . self::$foreground_colors[$colors[0]] . "m";
109         }
110         // 背景色
111         if(isset(self::$background_colors[$colors[1]])) {
112             $ret .= "\033[" . self::$background_colors[$colors[1]] . "m";
113         }
114         $ret .= $text;
